stripe.plans.create(
    {
        amount: 200,
        interval: 'month',
        name: 'Super Saver Plan',
        currency: 'usd',
        id: 'supsersaveplan200'
    },

    function(error, response) {

        console.log(response);

    }
);



//
//--------------------------
// .retrieve(plan_id, cb)

stripe.plans.retrieve(<plan_id>, function(error, response) {

    console.log(response);

});




//
//--------------------------
// .update(plan_id, data, cb)

stripe.plans.update(<plan_id>, { name: 'Even More Superer' }, function(error, response) {

    console.log(response);

});




//
//--------------------------
// .del(plan_id, cb)

stripe.plans.del(<plan_id>, function(error, response) {

    console.log(response);

});
