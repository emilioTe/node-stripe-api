//--------------------------
// .create(data, cb)

stripe.recipients.create(
    {
        name: 'John Doe',
        type: 'individual'
    },

    function(error, response) {

        console.log(response);

    }
);



//
//--------------------------
// .retrieve(recipient_id, cb)

stripe.recipients.retrieve(<recipient_id>, function(error, response) {

    console.log(response);

});




//
//--------------------------
// .update(recipient_id, data, cb)

stripe.recipients.update(<recipient_id>, { description: 'A user' }, function(error, response) {

    console.log(response);

});




//
//--------------------------
// .del(recipient_id, cb)

stripe.recipients.del(<recipient_id>, function(error, response) {

    console.log(response);

});




//
//--------------------------
// .list([data,] cb)

stripe.recipients.list(function(error, response) {

    console.log(response);

});


// OR


stripe.recipients.list({ offset: 5 }, function(error, response) {

    console.log(response);

});
